        // --- DOCUMENT DOWNLOAD FUNCTIONS ---

        function downloadPdf(job, text) {

            const doc = new jsPDF();

            const margin = 15;

            const pageHeight = doc.internal.pageSize.height;

            const lines = doc.splitTextToSize(text, 180);

            let cursor = margin;


            lines.forEach(line => {

                if (cursor > pageHeight - margin) {

                    doc.addPage();

                    cursor = margin;

                }

                doc.text(line, margin, cursor);

                cursor += 7;

            });

            doc.save(`Cover-Letter-${job.company}.pdf`);

        }


        function downloadWord(job, text) {

            const paragraphs = text.split('\n').map(p => new window.docx.Paragraph({

                children: [new window.docx.TextRun(p)],

            }));


            const doc = new window.docx.Document({

                sections: [{

                    properties: {},

                    children: paragraphs,

                }],

            });


            window.docx.Packer.toBlob(doc).then(blob => {

                const url = window.URL.createObjectURL(blob);

                const a = document.createElement("a");

                a.href = url;

                a.download = `Cover-Letter-${job.company}.docx`;

                a.click();

                window.URL.revokeObjectURL(url);

            });

        }

            }

            if (query) searchJobs(query, true);

        });


        function searchJobs(query, showLoader = true) {

            if(showLoader) { document.getElementById('loading-indicator').classList.remove('hidden'); }

            jobListingsContainer.innerHTML = '';

            document.getElementById('results-summary').classList.add('hidden');

            document.getElementById('no-results').classList.add('hidden');

            setTimeout(() => {

                const searchTerms = query.toLowerCase().split(/[\s,]+/);

                const results = mockJobs.filter(job => {

                    const jobText = `${job.title} ${job.company} ${job.location} ${job.description} ${job.tags.join(' ')}`.toLowerCase();

                    return searchTerms.every(term => jobText.includes(term));

                });

                if(showLoader) { document.getElementById('loading-indicator').classList.add('hidden'); }

                displayJobs(results, query);

            }, 1500);

        }


        function displayJobs(jobs, query) {

            if (jobs.length === 0) { document.getElementById('no-results').classList.remove('hidden'); return; }

            const summaryH2 = document.querySelector('#results-summary h2');

            summaryH2.textContent = `Found ${jobs.length} jobs matching your search for "${query}"`;

            document.getElementById('results-summary').classList.remove('hidden');

            jobListingsContainer.innerHTML = '';

            jobs.forEach(job => {

                const jobCard = createJobCard(job);

                jobListingsContainer.appendChild(jobCard);

            });

        }

        async function fetchSavedJobs(userId) {

            const savedJobsCollection = collection(db, `users/${userId}/savedJobs`);

            const querySnapshot = await getDocs(savedJobsCollection);

            savedJobIds.clear();

            querySnapshot.forEach((doc) => savedJobIds.add(doc.id));

        }


        async function toggleSaveJob(jobId, jobData) {

            if (!currentUser) return;

            const jobRef = doc(db, `users/${currentUser.uid}/savedJobs`, jobId);

            const saveButton = document.querySelector(`.save-job-btn[data-job-id="${jobId}"] svg`);

            if (savedJobIds.has(jobId)) {

                await deleteDoc(jobRef);

                savedJobIds.delete(jobId);

                saveButton.classList.remove('text-blue-600', 'fill-current');

                saveButton.classList.add('text-gray-400');

            } else {

                await setDoc(jobRef, jobData);

                savedJobIds.add(jobId);

                saveButton.classList.add('text-blue-600', 'fill-current');

                saveButton.classList.remove('text-gray-400');

            }

        }

        

        function displaySavedJobs() {

            savedJobListingsContainer.innerHTML = '';

            if (savedJobIds.size === 0) { noSavedJobsMessage.classList.remove('hidden'); return; }

            noSavedJobsMessage.classList.add('hidden');

            const allSavedJobs = mockJobs.filter(job => savedJobIds.has(job.id));

            allSavedJobs.forEach(job => {

                const jobCard = createJobCard(job);

                savedJobListingsContainer.appendChild(jobCard);

            });

        }
